# Note   : This is an example installation script. It shows how to setup a default BFD
# family installation. 
#
# Syntax notes:
# 
# Comments - any line that starts with a # is treated as a comment and ignored
# White space is ignored, as are tabs etc
#
# Each document is split in to two parts, head and body.
# The head of the document stores all of the setup information, such as the product id and they
# registry paths that are to be dealt with
# The body stores all the run time information, such as strings, position of controls etc
#
# $Id:$
#
# ===================================================================================
<fxpansion_installer_script>
    <head>
        # Currently 1000 is the only version. May be changed in future
        <script_version>1000</script_version>
        # The product family defines the type of install.
        # If the family is 'BFD' then the installation will show a kit options window
        # Otherwise the user gets to choose the install location only window
        <product_family>BFD</product_family>
        # The product id
        <product_id>FXJNF</product_id>
        # The product version
        <product_version>100</product_version>
        # The string that is displayed on the installer window
        <product_string>BFD Jazz & Funk</product_string>
        # Windows registry path that serial code is written and read from
        <product_registry_win>Software\fxpansion\BFD</product_registry_win>
        # Mac registry path that serial code is written and read from
        <product_registry_mac>com.FXpansion.BFD</product_registry_mac>
        # Product key to read from registry location
        <product_registry_key>productKeyJNF</product_registry_key>
        # What disk is this
        <disk_volume>3</disk_volume>
        # How many disks are there
        <disk_count>5</disk_count>
    </head>
    <body>
